                        vector res;
                
                        if(nums.empty()){
                            return res;
                        }
                
                        priority_queue, vector>, greater>> pq;
                        unordered_map m;
                        for(auto num : nums){
                            m[num]++;
                        }
                
                        for(auto &item : m){
                            pq.push({item.second, item.first});
                            if (pq.size() > k) {
                                pq.pop();
                            }
                        }        
                        
                        while(!pq.empty()) {
                            res.push_back(pq.top().second);
                            pq.pop();
                        }
                
                        reverse(res.begin(), res.end());
                        return res;
                    }
